    'column'       -- 租户ID字段隔离

    t.id as tenant_id,
    t.name as tenant_name,
    t.code as tenant_code,
    td.database_name,

    td.host,

-- 分片路由表
CREATE TABLE shard_routing (
    id VARCHAR(36) PRIMARY KEY DEFAULT gen_random_uuid(),
    table_name VARCHAR(100) NOT NULL,
    shard_key VARCHAR(255) NOT NULL,
    shard_id INTEGER NOT NULL,
    database_name VARCHAR(100) NOT NULL,
    schema_name VARCHAR(100) NOT NULL,
    table_suffix VARCHAR(50) NOT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(table_name, shard_key, shard_id)
);

-- 分片管理函数
CREATE OR REPLACE FUNCTION get_shard_info(
    p_table_name VARCHAR,
    p_shard_key VARCHAR
) RETURNS TABLE(
    shard_id INTEGER,
    database_name VARCHAR,
    schema_name VARCHAR,
    physical_table_name VARCHAR
) AS $$
DECLARE
    v_config sharding_config;
    v_shard_id INTEGER;
BEGIN
    -- 获取分片配置
    SELECT * INTO v_config 
    FROM sharding_config 
    WHERE table_name = p_table_name;
    
    -- 计算分片ID
    IF v_config.sharding_algorithm = 'hash' THEN
        v_shard_id := abs(hashtext(p_shard_key)) % v_config.shards_count;
    ELSIF v_config.sharding_algorithm = 'time' THEN
        -- 时间分片逻辑
        v_shard_id := extract(month FROM p_shard_key::TIMESTAMPTZ);
    ELSE
        v_shard_id := 0;
    END IF;
    
    -- 返回分片信息
    RETURN QUERY
    SELECT 
        sr.shard_id,
        sr.database_name,
        sr.schema_name,
        sr.table_suffix as physical_table_name
    FROM shard_routing sr
    WHERE sr.table_name = p_table_name 
      AND sr.shard_id = v_shard_id;
END;
$$ LANGUAGE plpgsql;

### 9.1.3 读写分离架构

```sql
-- 读写分离配置
CREATE TABLE database_clusters (
    id VARCHAR(36) PRIMARY KEY DEFAULT gen_random_uuid(),
    cluster_name VARCHAR(100) NOT NULL UNIQUE,
    cluster_type cluster_type NOT NULL,
    write_endpoint VARCHAR(255) NOT NULL,
    read_endpoints JSONB NOT NULL DEFAULT '[]',
    load_balance_strategy lb_strategy NOT NULL DEFAULT 'round_robin',
    health_check_config JSONB NOT NULL DEFAULT '{}',
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TYPE cluster_type AS ENUM (
    'primary',     -- 主集群
    'replica',     -- 只读副本
    'backup'       -- 备份集群
);

CREATE TYPE lb_strategy AS ENUM (
    'round_robin',
    'weighted',
    'least_connections',
    'latency_based'
);

-- 数据库节点配置
CREATE TABLE database_nodes (
    id VARCHAR(36) PRIMARY KEY DEFAULT gen_random_uuid(),
    cluster_id VARCHAR(36) NOT NULL REFERENCES database_clusters(id),
    endpoint VARCHAR(255) NOT NULL,
    node_role node_role NOT NULL,
    weight INTEGER NOT NULL DEFAULT 100,
    max_connections INTEGER NOT NULL DEFAULT 100,
    status node_status NOT NULL DEFAULT 'active',
    last_health_check TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(cluster_id, endpoint)
);

CREATE TYPE node_role AS ENUM (
    'primary',
    'secondary',
    'read_replica',
    'backup'
);

CREATE TYPE node_status AS ENUM (
    'active',
    'standby',
    'maintenance',
    'failed'
);

-- 读写分离路由函数
CREATE OR REPLACE FUNCTION get_read_write_endpoints(
    p_cluster_name VARCHAR,
    p_operation operation_type DEFAULT 'read'
) RETURNS TABLE(
    endpoint VARCHAR,
    node_role node_role,
    weight INTEGER
) AS $$
BEGIN
    IF p_operation = 'write' THEN
        -- 写操作只返回主节点
        RETURN QUERY
        SELECT 
            dn.endpoint,
            dn.node_role,
            dn.weight
        FROM database_nodes dn
        JOIN database_clusters dc ON dn.cluster_id = dc.id
        WHERE dc.cluster_name = p_cluster_name
          AND dn.node_role = 'primary'
          AND dn.status = 'active';
    ELSE
        -- 读操作返回所有活跃的读节点
        RETURN QUERY
        SELECT 
            dn.endpoint,
            dn.node_role,
            dn.weight
        FROM database_nodes dn
        JOIN database_clusters dc ON dn.cluster_id = dc.id
        WHERE dc.cluster_name = p_cluster_name
          AND dn.node_role IN ('secondary', 'read_replica')
          AND dn.status = 'active'
        ORDER BY 
            CASE dc.load_balance_strategy
                WHEN 'weighted' THEN -dn.weight  -- 权重降序
                ELSE dn.created_at
            END;
    END IF;
END;
$$ LANGUAGE plpgsql;

CREATE TYPE operation_type AS ENUM (
    'read',
    'write'
);

```sql
-- 数据备份策略配置
CREATE TABLE backup_strategies (
    id VARCHAR(36) PRIMARY KEY DEFAULT gen_random_uuid(),
    strategy_name VARCHAR(100) NOT NULL UNIQUE,
    backup_type backup_type NOT NULL,
    schedule_cron VARCHAR(50) NOT NULL,

    compression_enabled BOOLEAN NOT NULL DEFAULT true,
    encryption_enabled BOOLEAN NOT NULL DEFAULT true,
    storage_config JSONB NOT NULL DEFAULT '{}',
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TYPE backup_type AS ENUM (
    'full',
    'incremental',
    'differential',
    'continuous'
);

-- 备份任务执行记录
CREATE TABLE backup_executions (
    id VARCHAR(36) PRIMARY KEY DEFAULT gen_random_uuid(),
    strategy_id VARCHAR(36) NOT NULL REFERENCES backup_strategies(id),
    execution_type execution_type NOT NULL,
    start_time TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    end_time TIMESTAMPTZ,
    status execution_status NOT NULL DEFAULT 'running',
    backup_size BIGINT,
    storage_location VARCHAR(500),
    checksum VARCHAR(64),
    error_message TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TYPE execution_type AS ENUM (
    'scheduled',
    'manual',
    'emergency'
);

CREATE TYPE execution_status AS ENUM (
    'running',
    'completed',
    'failed',
    'cancelled'
);

-- 数据恢复记录
CREATE TABLE recovery_operations (
    id VARCHAR(36) PRIMARY KEY DEFAULT gen_random_uuid(),
    backup_execution_id VARCHAR(36) REFERENCES backup_executions(id),
    target_database VARCHAR(100) NOT NULL,
    target_timestamp TIMESTAMPTZ NOT NULL,
    recovery_type recovery_type NOT NULL,
    start_time TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    end_time TIMESTAMPTZ,
    status execution_status NOT NULL DEFAULT 'running',
    recovered_tables TEXT[],
    error_message TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TYPE recovery_type AS ENUM (
    'full',
    'partial',
    'point_in_time'
);

-- 备份监控视图
CREATE OR REPLACE VIEW backup_monitoring AS
SELECT 
    bs.strategy_name,
    bs.backup_type,
    bs.schedule_cron,
    be.execution_type,
    be.start_time,
    be.end_time,
    be.status,
    be.backup_size,
    be.storage_location,
    age(be.end_time, be.start_time) as duration,
    CASE 
        WHEN be.status = 'completed' THEN 'healthy'
        WHEN be.status = 'running' AND age(NOW(), be.start_time) > '1 hour'::interval THEN 'stalled'
        WHEN be.status = 'failed' THEN 'failed'
        ELSE 'warning'
    END as health_status
FROM backup_strategies bs
LEFT JOIN backup_executions be ON bs.id = be.strategy_id
WHERE be.start_time >= (NOW() - INTERVAL '7 days')
ORDER BY be.start_time DESC;

### 9.3.1 JPA实体配置

```java
// 基础实体配置
@MappedSuperclass
@EntityListeners(AuditingEntityListener.class)
@Data
public abstract class BaseEntity implements Serializable {
    
    @Id
    @GeneratedValue(generator = "UUID")
    @GenericGenerator(name = "UUID", strategy = "org.hibernate.id.UUIDGenerator")
    @Column(name = "id", updatable = false, nullable = false)
    private String id;
    
    @CreatedDate
    @Column(name = "created_at", updatable = false, nullable = false)
    private Date createdAt;
    
    @LastModifiedDate
    @Column(name = "updated_at", nullable = false)
    private Date updatedAt;
    
    @CreatedBy
    @Column(name = "created_by", updatable = false)
    private String createdBy;
    
    @LastModifiedBy
    @Column(name = "updated_by")
    private String updatedBy;
    
    @Version
    @Column(name = "version", nullable = false)

```java
// 缓存解决方案服务
@Service
@Slf4j
public class CacheSolutionService {
    
    private final RedisTemplate<String, Object> redisTemplate;
    private final RedissonClient redissonClient;
    
    public CacheSolutionService(RedisTemplate<String, Object> redisTemplate,
                              RedissonClient redissonClient) {
        this.redisTemplate = redisTemplate;
        this.redissonClient = redissonClient;
    }
    
    // 缓存穿透解决方案：布隆过滤器 + 空值缓存
    public <T> T getWithPenetrationProtection(String key, Class<T> type, 
                                            Supplier<T> loader, 
                                            BloomFilter<String> bloomFilter) {
        // 1. 检查布隆过滤器
        if (!bloomFilter.mightContain(key)) {
            log.debug("Key不在布隆过滤器中: {}", key);
            return null;
        }
        
        // 2. 从缓存获取
        T value = getFromCache(key, type);
        if (value != null) {
            return value;
        }
        
        // 3. 从数据源加载
        value = loader.get();
        if (value == null) {
            // 缓存空值，防止穿透
            setCacheWithNullProtection(key, null);
            return null;
        }
        
        // 4. 更新缓存
        setCache(key, value);
        return value;
    }
    
    // 缓存击穿解决方案：分布式锁
    public <T> T getWithBreakdownProtection(String key, Class<T> type, 
                                          Supplier<T> loader, 
                                          Duration lockTimeout) {
        // 1. 从缓存获取
        T value = getFromCache(key, type);
        if (value != null) {
            return value;
        }
        
        // 2. 获取分布式锁
        String lockKey = "lock:" + key;
        RLock lock = redissonClient.getLock(lockKey);
        
        try {
            // 尝试获取锁，最多等待5秒，锁超时时间可配置
            boolean locked = lock.tryLock(5, lockTimeout.toSeconds(), TimeUnit.SECONDS);
            if (!locked) {
                // 获取锁失败，直接返回null或降级处理
                return getFromCache(key, type); // 再次尝试，可能其他线程已经加载完成
            }
            
            // 双重检查，防止重复加载
            value = getFromCache(key, type);
            if (value != null) {
                return value;
            }
            
            // 3. 从数据源加载
            value = loader.get();
            if (value != null) {
                setCache(key, value);
            }
            
            return value;
            
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new RuntimeException("获取分布式锁被中断", e);
        } finally {
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }
    }
    
    // 缓存雪崩解决方案：随机过期时间 + 热点数据永不过期
    public <T> void setCacheWithAvalancheProtection(String key, T value, 
                                                  Duration baseTtl, 
                                                  boolean isHotData) {
        if (isHotData) {
            // 热点数据永不过期，但需要后台定期更新
            setCache(key, value);
            
            // 启动后台任务定期更新
            scheduleHotDataRefresh(key, value.getClass());
        } else {
            // 普通数据设置随机过期时间，避免同时失效
            Duration randomTtl = baseTtl.plus(Duration.ofSeconds(ThreadLocalRandom.current().nextInt(300)));
            setCacheWithTtl(key, value, randomTtl);
        }
    }
    
    // 热点数据后台刷新
    private <T> void scheduleHotDataRefresh(String key, Class<T> type) {
        // 使用Redis的过期事件或定时任务来刷新热点数据
        // 这里简化为设置一个标记
        redisTemplate.opsForValue().set("refresh:" + key, "1", Duration.ofHours(1));
    }
    
    // 缓存降级策略
    public <T> T getWithFallback(String key, Class<T> type, 
                               Supplier<T> primaryLoader,
                               Supplier<T> fallbackLoader) {
        try {
            return getWithBreakdownProtection(key, type, primaryLoader, Duration.ofSeconds(30));
        } catch (Exception e) {
            log.warn("主缓存策略失败，尝试降级策略: {}", key, e);
            return fallbackLoader.get();
        }
    }
    
    private <T> T getFromCache(String key, Class<T> type) {
        Object value = redisTemplate.opsForValue().get(key);
        if (value == null) {
            return null;
        }
        
        // 检查是否是空值标记
        if (value instanceof String && "NULL".equals(value)) {
            return null;
        }
        
        return type.cast(value);
    }
    
    private <T> void setCache(String key, T value) {
        redisTemplate.opsForValue().set(key, value);
    }
    
    private <T> void setCacheWithTtl(String key, T value, Duration ttl) {
        redisTemplate.opsForValue().set(key, value, ttl);
    }
    
    private void setCacheWithNullProtection(String key, Object value) {
        // 空值缓存5分钟，防止缓存穿透
        redisTemplate.opsForValue().set(key, "NULL", Duration.ofMinutes(5));
    }

        }

    @WriteTransaction
    public <T> void updateWithWriteThrough(String key, T newValue, Runnable databaseUpdate) {
        try {
            // 1. 更新缓存
            setCache(key, newValue);
            
            // 2. 更新数据库
            databaseUpdate.run();
            
        } catch (Exception e) {
            // 回滚缓存
            deleteCache(key);
            log.error("Write-Through模式更新失败: {}", key, e);
            throw e;
        }
    }

    @EventListener
    public void handleDatabaseChange(DatabaseChangeEvent event) {
        String tableName = event.getTableName();
        String operation = event.getOperation();
        String rowId = event.getRowId();
        
        // 根据表名和操作类型生成缓存key
        String cacheKey = generateCacheKey(tableName, rowId);
        
        switch (operation) {
            case "INSERT":
            case "UPDATE":
                // 删除缓存，下次查询时重新加载
                deleteCache(cacheKey);
                break;
            case "DELETE":
                // 直接删除缓存
                deleteCache(cacheKey);
                break;
            default:
                log.warn("未知的数据库操作: {}", operation);
        }
        
        // 发送缓存失效事件
        sendCacheEvictEvent(cacheKey);
    }
    
    // 延迟双删策略，解决数据库主从延迟问题
    @WriteTransaction
    public <T> void updateWithDoubleDelete(String key, T newValue, Runnable databaseUpdate) {
        try {
            // 1. 第一次删除缓存
            deleteCache(key);
            
            // 2. 更新数据库
            databaseUpdate.run();
            
            // 3. 延迟第二次删除缓存（等待主从同步）
            scheduleDelayedDelete(key, Duration.ofSeconds(1));
            
        } catch (Exception e) {
            log.error("延迟双删策略更新失败: {}", key, e);
            throw e;
        }
    }

    public <T> void setCacheWithVersion(String key, T value, String version) {
        String versionedKey = key + ":v" + version;
        setCache(versionedKey, value);
        
        // 更新当前版本的指针
        redisTemplate.opsForValue().set(key + ":current", version);

        // 获取当前版本
        String currentVersion = (String) redisTemplate.opsForValue().get(key + ":current");
        if (currentVersion == null) {
            return null;
        }
        
        String versionedKey = key + ":v" + currentVersion;
        return getFromCache(versionedKey, type);
    }

        log.debug("删除缓存: {}", key);
    }

        redisTemplate.opsForValue().set(key, value);
    }
    
    private <T> T getFromCache(String key, Class<T> type) {
        Object value = redisTemplate.opsForValue().get(key);
        return value != null ? type.cast(value) : null;
    }

        kafkaTemplate.send("cache.evict", event);
    }

        CompletableFuture.delayedExecutor(delay.toMillis(), TimeUnit.MILLISECONDS)
            .execute(() -> deleteCache(key));
    }
    
    private String generateCacheKey(String tableName, String rowId) {
        return String.format("cache:%s:%s", tableName, rowId);
    }

    }

    }

            }
        }

    private void handleChangeEvent(ChangeEvent<String, String> changeEvent) {
        String key = changeEvent.key();
        String value = changeEvent.value();
        
        log.debug("收到数据库变更事件 - Key: {}, Value: {}", key, value);
        
        // 解析变更事件
        DatabaseChangeEvent event = parseChangeEvent(value);
        if (event != null) {
            // 发布Spring事件
            ApplicationEventPublisher.publishEvent(event);
        }
    }
    
    private DatabaseChangeEvent parseChangeEvent(String jsonValue) {
        try {
            JsonNode jsonNode = objectMapper.readTree(jsonValue);
            JsonNode payload = jsonNode.get("payload");
            
            if (payload != null) {
                String operation = payload.get("op").asText();
                String tableName = payload.get("source").get("table").asText();
                String rowId = extractRowId(payload);
                
                return new DatabaseChangeEvent(tableName, operation, rowId);
            }
        } catch (Exception e) {
            log.error("解析数据库变更事件失败", e);
        }
        return null;
    }

@Service
@Slf4j
public class VersionRollbackService {
    
    private final Flyway flyway;
    private final MigrationHistoryRepository migrationHistoryRepository;
    private final DatabaseBackupService backupService;
    
    public VersionRollbackService(Flyway flyway,
                                MigrationHistoryRepository migrationHistoryRepository,
                                DatabaseBackupService backupService) {
        this.flyway = flyway;
        this.migrationHistoryRepository = migrationHistoryRepository;
        this.backupService = backupService;
    }
    
    // 数据库版本回滚
    @Transactional
    public RollbackResult rollbackToVersion(String targetVersion, boolean createBackup) {
        RollbackResult result = new RollbackResult();
        result.setStartTime(LocalDateTime.now());
        
        try {
            log.info("开始回滚到版本: {}", targetVersion);
            
            // 1. 创建备份
            if (createBackup) {
                String backupId = backupService.createBackup("rollback_" + targetVersion);
                result.setBackupId(backupId);
            }
            
            // 2. 执行回滚
            int rolledBackMigrations = flyway.rollbackToVersion(targetVersion);
            result.setRolledBackMigrations(rolledBackMigrations);
            
            // 3. 验证回滚结果
            boolean success = validateRollback(targetVersion);
            result.setSuccess(success);
            
            if (success) {
                log.info("版本回滚成功: {} -> {}", getCurrentVersion(), targetVersion);
            } else {
                log.error("版本回滚验证失败");
            }
            
        } catch (Exception e) {
            log.error("版本回滚失败", e);
            result.setSuccess(false);
            result.setErrorMessage(e.getMessage());
            
            // 尝试恢复备份
            if (createBackup && result.getBackupId() != null) {
                restoreFromBackup(result.getBackupId());
            }
        } finally {
            result.setEndTime(LocalDateTime.now());
            saveRollbackHistory(result);
        }
        
        return result;
    }
    
    // 数据回滚（业务数据）
    @Transactional
    public DataRollbackResult rollbackData(LocalDateTime targetTime, 
                                         Set<String> tablesToRollback) {
        DataRollbackResult result = new DataRollbackResult();
        result.setTargetTime(targetTime);
        
        try {
            log.info("开始数据回滚到时间点: {}", targetTime);
            
            // 1. 备份当前数据
            String backupId = backupService.createBackup("data_rollback_" + 
                targetTime.format(DateTimeFormatter.ISO_LOCAL_DATE_TIME));
            result.setBackupId(backupId);
            
            // 2. 执行数据回滚
            for (String tableName : tablesToRollback) {
                int affectedRows = rollbackTableData(tableName, targetTime);
                result.addTableResult(tableName, affectedRows);
            }
            
            // 3. 验证数据一致性
            boolean consistent = validateDataConsistency(targetTime, tablesToRollback);
            result.setSuccess(consistent);
            
            log.info("数据回滚完成: {}", result);
            
        } catch (Exception e) {
            log.error("数据回滚失败", e);
            result.setSuccess(false);
            result.setErrorMessage(e.getMessage());
            
            // 恢复备份
            if (result.getBackupId() != null) {
                restoreFromBackup(result.getBackupId());
            }
        }
        
        return result;
    }

    public EmergencyRollbackResult emergencyRollback(String rollbackPlanId) {
        EmergencyRollbackResult result = new EmergencyRollbackResult();
        
        try {
            log.warn("执行紧急回滚: {}", rollbackPlanId);
            
            // 1. 停止应用服务
            stopApplicationServices();

        } catch (Exception e) {

            result.setSuccess(false);

SELECT 
    table_name,
    metric_name,
    AVG(metric_value) as avg_value,
    MIN(metric_value) as min_value,
    MAX(metric_value) as max_value,
    COUNT(*) as check_count,

                    triggerQualityAlert(rule, metric);
                }
                
            } catch (Exception e) {

            }
        }
        
        log.info("数据质量检查完成");
    }
